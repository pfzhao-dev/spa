procedure First00 {
  read x00;
  read z00;
  call Second00; }
 
  procedure Second00 {
  x00 = 0;
  i00 = 5;
  while (i00!=0) {
      x00 = x00 + 2*y00;
      call Third00;
      i00 = i00 - 1; }
  if (x00==1) then {
      x00 = x00+1; }
      else {
      z00 = 1; }
  z00 = z00 + x00 + i00;
  y00 = z00 + 2;
  x00 = x00 * y00 + z00; }
 
  procedure Third00 {
      z00 = 5;
      v00 = z00;
      print v00; }
